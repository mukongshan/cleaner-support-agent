```typescript
const handleSubmitTicket = async () => {
  try {
    setTicketSubmitting(true);
    setTicketError(null);

    const result = await createTicket({
      title: ticketFormData.problemSummary || '工单',
      description: `问题类型: ${getProblemTypeLabel(ticketFormData.problemType)}\n...`,
      priority: ticketFormData.priority,
      relatedChatId: currentSessionId || undefined,
      attachmentUrls: ticketFormData.images.length > 0 ? ticketFormData.images : undefined
    });

    setShowTicketForm(false);
    setTicketCreated(true);
    
    if (onCreateTicket) {
      onCreateTicket();
    }
  } catch (err: any) {
    setTicketError(err.message || '创建工单失败');
  } finally {
    setTicketSubmitting(false);
  }
};
```

## 5. 🎯 使用流程

### 问答界面创建工单：

1. **用户进行对话**（有对话记录）
2. **点击右上角 "+" → "创建工单"**
3. **看到统一的工单表单**
4. **可选：点击 "AI 辅助填写" 按钮**
   - 系统分析对话记录
   - 自动填充表单字段
   - 用户可以修改填充的内容
5. **填写/修改表单信息**
6. **点击"确认创建工单"**
7. **看到成功提示**

### 工单界面创建工单：

1. **点击右上角 "+" 按钮**
2. **看到统一的工单表单**
3. **没有 AI 辅助按钮**（因为没有对话上下文）
4. **手动填写表单信息**
5. **点击"确认创建工单"**
6. **看到成功提示并刷新列表**

## 7. ✨ AI 辅助填写的智能提取

### 提取的信息：
- **问题类型**：根据关键词判断（故障/维护/咨询/投诉）
- **优先级**：根据严重程度推荐（低/中/高）
- **问题摘要**：提取第一条用户消息作为标题
- **关键对话**：提取重要的用户消息
- **图片**：提取对话中上传的图片

### 提取逻辑：
```typescript
const extractInfoFromChat = (): AIExtractedInfo => {
  const userMessages = messages.filter(m => m.type === 'user');
  
  // 提取问题摘要
  const firstUserMsg = userMessages[0]?.content || '';
  const problemSummary = firstUserMsg.length > 50
    ? firstUserMsg.substring(0, 50) + '...'
    : firstUserMsg;

  // 分析问题类型和优先级
  const allUserContent = userMessages.map(m => m.content.toLowerCase()).join(' ');
  
  let suggestedType = 'maintenance';
  let suggestedPriority: 'low' | 'medium' | 'high' = 'medium';
  
  if (allUserContent.includes('故障') || allUserContent.includes('不转')) {
    suggestedType = 'malfunction';
    suggestedPriority = 'high';
  } else if (allUserContent.includes('如何') || allUserContent.includes('怎么')) {
    suggestedType = 'consultation';
    suggestedPriority = 'low';
  }
  
  return {
    problemSummary,
    suggestedType,
    suggestedPriority,
    keyDialogues: userMessages.map(m => m.content),
    images: messages.filter(m => m.image).map(m => m.image!)
  };
};
```
